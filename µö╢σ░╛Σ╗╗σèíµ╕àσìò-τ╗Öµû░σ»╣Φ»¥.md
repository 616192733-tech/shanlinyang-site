## WebP批量转换脚本(第4步用,实测省55%体积)
```python
from PIL import Image
import os, glob, re

# 1) 转换所有 jpg/png 为 webp
converted = {}
for f in glob.glob("img/**/*.jpg", recursive=True) + glob.glob("img/**/*.png", recursive=True):
    im = Image.open(f).convert("RGB")
    webp = f.rsplit(".",1)[0] + ".webp"
    im.save(webp, "WEBP", quality=82, method=6)
    converted[f] = webp
    os.remove(f)  # 删除原图
    print(f"{os.path.basename(f)} -> webp")

# 2) 全站HTML替换 .jpg/.png 引用为 .webp
for html in glob.glob("*.html") + glob.glob("**/*.html", recursive=True):
    s = open(html).read()
    s = re.sub(r'(/img/[^"\']*?)\.(jpg|png)', r'\1.webp', s)
    open(html, "w").write(s)
print("全站HTML引用已更新为webp")
```
